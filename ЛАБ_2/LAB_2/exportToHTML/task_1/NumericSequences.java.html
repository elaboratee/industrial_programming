package task_1;

import java.util.Scanner;

/**
 * Class that represents methods for working with numeric sequences
 * @author Ilya Sokol
 */
class NumericSequences {
    /**
     * Method that generates an array of n integers entered into the console
     */
    public static int[] getIntegerSequence() {
        Scanner console = new Scanner(System.in);

        System.out.print("Введите количество чисел: ");
        int n = console.nextInt();
        int[] numArray = new int[n];

        for (int i = 0; i < n; i++) {
            System.out.printf("Введите %d-е число: ", i + 1);
            numArray[i] = console.nextInt();
        }
        console.close();
        return numArray;
    }

    /**
     * Method that searches for shortest and longest integers in an array
     * and prints it into the console
     * @param numArray array of integers
     */
    public static void findShortestAndLongest(int[] numArray) {
        int strLength;
        int minLength = Integer.MAX_VALUE;
        int maxLength = 0;
        int shortNumber = 0, longNumber = 0;
        String strNumber;

        for (int number : numArray) {
            strNumber = String.valueOf(number).replaceAll("-", "");
            strLength = strNumber.length();

            if (strLength < minLength) {
                minLength = strLength;
                shortNumber = number;
            }
            if (strLength > maxLength) {
                maxLength = strLength;
                longNumber = number;
            }
        }
        System.out.printf("\nСамое короткое число = %d\t\tЕго длина = %d\n", shortNumber, minLength);
        System.out.printf("Самое длинное число = %d\t\tЕго длина = %d\n", longNumber, maxLength);
    }
}
